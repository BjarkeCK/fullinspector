76  public object CreateInstance() {
77  // Unity requires special construction logic for types that derive from
78  // ScriptableObject. The normal inspector reflection logic will create ScriptableObject
79  // instances if FullInspectorSettings.AutomaticReferenceInstantation has been set to
80  // true.
81  if (typeof(ScriptableObject).IsAssignableFrom(ReflectedType)) {
82  return ScriptableObject.CreateInstance(ReflectedType);
83  }
84 
85  if (HasDefaultConstructor == false) {
86  return FormatterServices.GetSafeUninitializedObject(ReflectedType);
87  }
88 
89  if (_isArray) {
90  // we have to start with a size zero array otherwise it will have invalid data
91  // inside of it
92  return Array.CreateInstance(ElementType, 0);
93  }
94 
95  try {
96  return Activator.CreateInstance(ReflectedType, /*nonPublic:*/ true);
97  }
98  catch (MissingMethodException e) {
99  throw new InvalidOperationException("Unable to create instance of " + ReflectedType + "; there is no default constructor", e);
100  }
101  catch (TargetInvocationException e) {
102  throw new InvalidOperationException("Constructor of " + ReflectedType + " threw an exception when creating an instance", e);
103  }
104  catch (MemberAccessException e) {
105  throw new InvalidOperationException("Unable to access constructor of " + ReflectedType, e);
106  }
107  }

113  internal InspectedType(Type type) {
114  ReflectedType = type;
115 
116  // determine if we are a collection or array; recall that arrays implement the
117  // ICollection interface, however
118 
119  _isArray = type.IsArray;
120  IsCollection = _isArray || type.IsImplementationOf(typeof(ICollection<>));
121 
122  // If we're a collection or array, get the generic type definition so that client code
123  // can determine how to deserialize child elements
124  if (_isArray == false && IsCollection) {
125  Type collectionType = type.GetInterface(typeof(ICollection<>));
126  _elementType = collectionType.GetGenericArguments()[0];
127  }
128  else if (_isArray) {
129  _elementType = type.GetElementType();
130  }
131 
132  // If we're not one of those three types, then we will be using Properties to assign
133  // data to ourselves, so we want to lookup said information
134  else {
135  // Lookup the local properties
136  _localInspectedProperties = CollectLocalProperties(type).OrderBy(property =>
137  InspectorOrderAttribute.GetInspectorOrder(property.MemberInfo)
138  ).ToList();
139 
140  // Find all of the properties on this object (and on parent types)
141  _inspectedProperties = new List<InspectedProperty>(_localInspectedProperties);
142  if (ReflectedType.BaseType != null) {
143  // We go through InspectedType for parent types so that removing a property from
144  // a parent type will correctly propagate down.
145  var inspectedParentType = InspectedType.Get(ReflectedType.BaseType);
146  _inspectedProperties.AddRange(inspectedParentType._inspectedProperties);
147  }
148 
149  // Sort the properties by their order. We use OrderBy instead of Sort because
150  // OrderBy is guaranteed to be stable.
151  _inspectedProperties = _inspectedProperties.OrderBy(property =>
152  InspectorOrderAttribute.GetInspectorOrder(property.MemberInfo)
153  ).ToList();
154 
155 
156  // Get the list of properties that we serialize.
157  _serializedProperties = (from property in _inspectedProperties
158  where IsSerializedByFullInspector(property)
159  where IsSerializedByUnity(property) == false
160  select property).ToList();
161 
162  // Add our property names in
163  _nameToProperty = new Dictionary<string, InspectedProperty>();
164  foreach (var property in _inspectedProperties) {
165  _nameToProperty[property.Name] = property;
166  }
167 
168  // Collect all of the methods that act as buttons for this type.
169  BindingFlags allMethodBindingFlags =
170  BindingFlags.Instance |
171  BindingFlags.Static |
172  BindingFlags.FlattenHierarchy |
173  BindingFlags.Public |
174  BindingFlags.NonPublic;
175 
176  BindingFlags localMethodBindingFlags =
177  BindingFlags.Instance |
178  BindingFlags.Static |
179  BindingFlags.DeclaredOnly |
180  BindingFlags.Public |
181  BindingFlags.NonPublic;
182 
183  // Find and order all of the buttons on the type.
184  _buttons = (from method in ReflectedType.GetMethods(allMethodBindingFlags)
185  where Attribute.IsDefined(method, typeof(InspectorButtonAttribute))
186  orderby InspectorOrderAttribute.GetInspectorOrder(method)
187  select new InspectedMethod(method)).ToArray();
188 
189  _localButtons = (from method in ReflectedType.GetMethods(localMethodBindingFlags)
190  where Attribute.IsDefined(method, typeof(InspectorButtonAttribute))
191  orderby InspectorOrderAttribute.GetInspectorOrder(method)
192  select new InspectedMethod(method)).ToArray();
193 
194  // Collect all of the methods on the type.
195  _methods = (from method in ReflectedType.GetMethods(allMethodBindingFlags)
196  where Attribute.IsDefined(method, typeof(HideInInspector)) == false
197  select new InspectedMethod(method)).ToArray();
198  }
199  }

206  private static bool IsInspectableProperty(PropertyInfo property) {
207  // We don't serialize delegates
208  if (typeof(Delegate).IsAssignableFrom(property.PropertyType)) {
209  return false;
210  }
211 
212  // If the property cannot be both read and written to, we don't serialize it
213  if (property.CanRead == false || property.CanWrite == false) {
214  return false;
215  }
216 
217  // If the property is named "Item", it might be the this[int] indexer, which in
218  // that case we don't serialize it We cannot just compare with "Item" because of
219  // explicit interfaces, where the name of the property will be the full method
220  // name.
221  if (property.Name.EndsWith("Item")) {
222  ParameterInfo[] parameters = property.GetIndexParameters();
223  if (parameters.Length > 0) {
224  return false;
225  }
226  }
227 
228  return true;
229  }

235  private static bool IsInspectableField(FieldInfo field) {
236  // We don't serialize delegates
237  if (typeof(Delegate).IsAssignableFrom(field.FieldType)) {
238  return false;
239  }
240 
241  // We don't serialize compiler generated fields (an example would be a backing
242  // field for an automatically generated property).
243  if (field.IsDefined(typeof(CompilerGeneratedAttribute), false)) {
244  return false;
245  }
246 
247  return true;
248  }

253  public IEnumerable<InspectedProperty> StaticProperties {
254  get {
255  if (IsCollection) {
256  yield break;
257  }

260  BindingFlags.NonPublic |
261  BindingFlags.Public |
262  BindingFlags.FlattenHierarchy |
263  BindingFlags.Static;
264 
265  foreach (MemberInfo member in ReflectedType.GetMembers(flags)) {
266  FieldInfo field = member as FieldInfo;
267  PropertyInfo prop = member as PropertyInfo;
268 
269  if (field != null && InspectedType.IsInspectableField(field)) {
270  yield return new InspectedProperty(field);
271  }
272 
273  if (prop != null && InspectedType.IsInspectableProperty(prop)) {
274  yield return new InspectedProperty(prop);
275  }
276  }
