584  private void DrawListItem(EventType eventType, Rect position, IReorderableListAdaptor adaptor, int itemIndex) {
585  bool visible = (position.y < _visibleRect.yMax && position.yMax > _visibleRect.y);
586  bool draggable = _allowReordering && adaptor.CanDrag(itemIndex);
587 
588  Rect itemContentPosition = position;
589  itemContentPosition.x = position.x + 2;
590  itemContentPosition.y += 1;
591  itemContentPosition.width = position.width - 4;
592  itemContentPosition.height = position.height - 4;
593 
594  // Make space for grab handle?
595  if (draggable) {
596  itemContentPosition.x += 20;
597  itemContentPosition.width -= 20;
598  }
599 
600  // Make space for element index.
601  if (_indexLabelWidth != 0) {
602  itemContentPosition.width -= _indexLabelWidth;
603 
604  if (eventType == EventType.Repaint && visible)
605  s_RightAlignedLabelStyle.Draw(new Rect(itemContentPosition.x, position.y, _indexLabelWidth, position.height - 4), itemIndex + ":", false, false, false, false);
606 
607  itemContentPosition.x += _indexLabelWidth;
608  }
609 
610  // Make space for remove button?
611  if (hasRemoveButtons)
612  itemContentPosition.width -= removeButtonStyle.fixedWidth;
613 
614  if (eventType == EventType.Repaint && visible) {
615  // Draw grab handle?
616  if (draggable)
617  GUI.DrawTexture(new Rect(position.x + 6, position.y + position.height / 2f - 3, 9, 5), ReorderableListResources.texGrabHandle);
618 
619  // Draw splitter between list items.
620  if (!_tracking || itemIndex != s_AnchorIndex)
621  GUI.DrawTexture(new Rect(position.x, position.y - 1, position.width, 1), ReorderableListResources.texItemSplitter);
622  }
623 
624  // Allow control to be automatically focused.
625  if (s_AutoFocusIndex == itemIndex)
626  GUI.SetNextControlName("AutoFocus_" + _controlID + "_" + itemIndex);
627 
628  try {
629  s_CurrentItemIndex.Push(itemIndex);
630 
631  // Present actual control.
632  EditorGUI.BeginChangeCheck();
633  adaptor.DrawItem(itemContentPosition, itemIndex);
634  if (EditorGUI.EndChangeCheck())
635  ReorderableListGUI.indexOfChangedItem = itemIndex;
636 
637  // Draw remove button?
638  if (hasRemoveButtons && adaptor.CanRemove(itemIndex)) {
639  s_RemoveButtonPosition = position;
640  s_RemoveButtonPosition.width = removeButtonStyle.fixedWidth;
641  s_RemoveButtonPosition.x = itemContentPosition.xMax + 2;
642  s_RemoveButtonPosition.height -= 2;
643 
644  if (DoRemoveButton(s_RemoveButtonPosition, visible))
645  RemoveItem(adaptor, itemIndex);
646  }
647 
648  // Check for context click?
649  if (eventType == EventType.ContextClick && position.Contains(Event.current.mousePosition) && (flags & ReorderableListFlags.DisableContextMenu) == 0) {
650  ShowContextMenu(_controlID, itemIndex, adaptor);
651  Event.current.Use();
652  }
653  }
654  finally {
655  s_CurrentItemIndex.Pop();
656  }
657  }

711  private void DrawListContainerAndItems(Rect position, int controlID, IReorderableListAdaptor adaptor) {
712  // Get local copy of event information for efficiency.
713  EventType eventType = Event.current.GetTypeForControl(controlID);
714  Vector2 mousePosition = Event.current.mousePosition;
715  if (Event.current.isMouse)
716  s_MousePosition = GUIUtility.GUIToScreenPoint(mousePosition);
717 
718  int newTargetIndex = s_TargetIndex;
719 
720  // Position of first item in list.
721  float firstItemY = position.y + containerStyle.padding.top;
722 
723  switch (eventType) {
724  case EventType.MouseDown:
725  if (_tracking) {
726  // Cancel drag when other mouse button is pressed.
727  s_TrackingCancelBlockContext = true;
728  Event.current.Use();
729  }
730  break;
731 
732  case EventType.MouseDrag:
733  if (_tracking) {
734  // Reset target index and adjust when looping through list items.
735  if (mousePosition.y < firstItemY)
736  newTargetIndex = 0;
737  else if (mousePosition.y >= position.yMax)
738  newTargetIndex = adaptor.Count;
739 
740  s_DragItemPosition.y = Mathf.Clamp(mousePosition.y + s_AnchorMouseOffset, firstItemY, position.yMax - s_DragItemPosition.height - 1);
741  }
742  break;
743 
744  case EventType.MouseUp:
745  if (controlID == GUIUtility.hotControl) {
746  // Allow user code to change control over reordering during drag.
747  if (!s_TrackingCancelBlockContext && _allowReordering)
748  AcceptReorderDrag(adaptor);
749  else
750  StopTrackingReorderDrag();
751  Event.current.Use();
752  }
753  break;
754 
755  case EventType.KeyDown:
756  if (_tracking && Event.current.keyCode == KeyCode.Escape) {
757  StopTrackingReorderDrag();
758  Event.current.Use();
759  }
760  break;
761 
762  case EventType.ExecuteCommand:
763  if (s_ContextControlID == controlID) {
764  int itemIndex = s_ContextItemIndex;
765  try {
766  DoCommand(s_ContextCommandName, itemIndex, adaptor);
767  Event.current.Use();
768  }
769  finally {
770  s_ContextControlID = 0;
771  s_ContextItemIndex = 0;
772  }
773  }
774  break;
775 
776  case EventType.Repaint:
777  // Draw caption area of list.
778  containerStyle.Draw(position, GUIContent.none, false, false, false, false);
779  break;
780  }
781 
782  ReorderableListGUI.indexOfChangedItem = -1;
783 
784  // Draw list items!
785  Rect itemPosition = new Rect(position.x + 2, firstItemY, position.width - 4, 0);
786  float targetSlotPosition = position.yMax - s_DragItemPosition.height - 1;
787 
788  float lastMidPoint = 0f;
789  float lastHeight = 0f;
790 
791  int count = adaptor.Count;
792  for (int i = 0; i < count; ++i) {
793  itemPosition.y = itemPosition.yMax;
794  itemPosition.height = 0;
795 
796  if (_tracking) {
797  // Does this represent the target index?
798  if (i == s_TargetIndex) {
799  targetSlotPosition = itemPosition.y;
800  itemPosition.y += s_DragItemPosition.height;
801  }
802 
803  // Do not draw item if it is currently being dragged. Draw later so that it is
804  // shown in front of other controls.
805  if (i == s_AnchorIndex)
806  continue;
807 
808  lastMidPoint = itemPosition.y - lastHeight / 2f;
809  }
810 
811  // Update position for current item.
812  itemPosition.height = adaptor.GetItemHeight(i) + 4;
813  lastHeight = itemPosition.height;
814 
815  if (_tracking && eventType == EventType.MouseDrag) {
816  float midpoint = itemPosition.y + itemPosition.height / 2f;
817 
818  if (s_TargetIndex < i) {
819  if (s_DragItemPosition.yMax > lastMidPoint && s_DragItemPosition.yMax < midpoint)
820  newTargetIndex = i;
821  }
822  else if (s_TargetIndex > i) {
823  if (s_DragItemPosition.y > lastMidPoint && s_DragItemPosition.y < midpoint)
824  newTargetIndex = i;
825  }
826 
827  /*if (s_DragItemPosition.y > itemPosition.y && s_DragItemPosition.y <= midpoint)
828  newTargetIndex = i;
829  else if (s_DragItemPosition.yMax > midpoint && s_DragItemPosition.yMax <= itemPosition.yMax)
830  newTargetIndex = i + 1;*/
831  }
832 
833  // The following may break use of tab key to navigate through controls :/
834  if ((flags & ReorderableListFlags.DisableClipping) == 0) {
835  // Clip list item? Performance boost!
836  if (itemPosition.yMax < _visibleRect.y - itemPosition.height) {
837  // Let's try and trick Unity into maintaining tab key support...
838  GUIUtility.GetControlID(FocusType.Keyboard, itemPosition);
839  continue;
840  }
841  if (itemPosition.y > _visibleRect.yMax + itemPosition.height)
842  break;
843  }
844 
845  // Draw list item.
846  DrawListItem(eventType, itemPosition, adaptor, i);
847 
848  // Did list count change (i.e. item removed)?
849  if (adaptor.Count < count) {
850  // We assume that it was this item which was removed, so --i allows us to
851  // process the next item as usual.
852  count = adaptor.Count;
853  --i;
854  continue;
855  }
856 
857  // Event has already been used, skip to next item.
858  if (Event.current.type != EventType.Used) {
859  switch (eventType) {
860  case EventType.MouseDown:
861  if (GUI.enabled && itemPosition.Contains(mousePosition)) {
862  // Remove input focus from control before attempting a context click
863  // or drag.
864  GUIUtility.keyboardControl = 0;
865 
866  if (_allowReordering && adaptor.CanDrag(i) && Event.current.button == 0) {
867  s_DragItemPosition = itemPosition;
868 
869  BeginTrackingReorderDrag(controlID, i);
870  s_AnchorMouseOffset = itemPosition.y - mousePosition.y;
871  s_TargetIndex = i;
872 
873  Event.current.Use();
874  }
875  }
876  break;
877  /* DEBUG
878  case EventType.Repaint:
879  GUI.color = Color.red;
880  GUI.DrawTexture(new Rect(0, lastMidPoint, 10, 1), EditorGUIUtility.whiteTexture);
881  GUI.color = Color.yellow;
882  GUI.DrawTexture(new Rect(5, itemPosition.y + itemPosition.height / 2f, 10, 1), EditorGUIUtility.whiteTexture);
883  GUI.color = Color.white;
884  break;
885  //*/
886  }
887  }
888  }
889 
890  // Item which is being dragged should be shown on top of other controls!
891  if (IsTrackingControl(controlID)) {
892  lastMidPoint = position.yMax - lastHeight / 2f;
893 
894  if (eventType == EventType.MouseDrag) {
895  if (s_DragItemPosition.yMax >= lastMidPoint)
896  newTargetIndex = count;
897 
898  // Force repaint to occur so that dragging rectangle is visible.
899  s_TargetIndex = newTargetIndex;
900  Event.current.Use();
901  }
902 
903  DrawFloatingListItem(eventType, adaptor, targetSlotPosition);
904  /* DEBUG
905  if (eventType == EventType.Repaint) {
906  GUI.color = Color.blue;
907  GUI.DrawTexture(new Rect(100, lastMidPoint, 20, 1), EditorGUIUtility.whiteTexture);
908  GUI.color = Color.white;
909  }
910  //*/
911  }
