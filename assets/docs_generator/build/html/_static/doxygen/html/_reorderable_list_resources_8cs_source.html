152  private static void LoadResourceAssets() {
153  var skin = EditorGUIUtility.isProSkin ? s_DarkSkin : s_LightSkin;
154  s_Cached = new Texture2D[skin.Length];
155 
156  for (int i = 0; i < s_Cached.Length; ++i) {
157  // Get image data (PNG) from base64 encoded strings.
158  byte[] imageData = Convert.FromBase64String(skin[i]);
159 
160  // Gather image size from image data.
161  int texWidth, texHeight;
162  GetImageSize(imageData, out texWidth, out texHeight);
163 
164  // Generate texture asset.
165  var tex = new Texture2D(texWidth, texHeight, TextureFormat.ARGB32, false);
166  tex.hideFlags = HideFlags.HideAndDontSave;
167  tex.name = "(Generated) ReorderableList:" + i;
168  tex.filterMode = FilterMode.Point;
169  tex.LoadImage(imageData);
170 
171  s_Cached[i] = tex;
172  }
173 
174  s_LightSkin = null;
175  s_DarkSkin = null;
176  }

184  private static void GetImageSize(byte[] imageData, out int width, out int height) {
185  width = ReadInt(imageData, 3 + 15);
186  height = ReadInt(imageData, 3 + 15 + 2 + 2);
187  }
188 
189  private static int ReadInt(byte[] imageData, int offset) {
190  return (imageData[offset] << 8) | imageData[offset + 1];
191  }
